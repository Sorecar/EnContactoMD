CREATE SCHEMA `encontactomd` ;

CREATE TABLE `encontactomd`.`usuarios` (
  `Id` INT NOT NULL AUTO_INCREMENT,
  `Usuario` VARCHAR(45) NOT NULL,
  `Contrasena` VARCHAR(45) NOT NULL,
  `NumTel` VARCHAR(45) NOT NULL,
  `NomMascota` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`Id`));

CREATE TABLE `encontactomd`.`noticias` (
  `IdNoticia` INT NOT NULL,
  `Usuario` VARCHAR(45) NOT NULL,
  `Noticia` VARCHAR(45) NOT NULL,
  `HoraNoticia` VARCHAR(45) NOT NULL,
  `FechaNoticia` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`IdNoticia`));

CREATE TABLE `encontactomd`.`eventos` (
  `IdEventos` INT NOT NULL AUTO_INCREMENT,
  `ConInfo` VARCHAR(45) NOT NULL,
  `FechaEvento` VARCHAR(45) NOT NULL,
  `HoraEvento` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`IdEventos`));
